‚ö° Quick Example

Build a Dashboard in 30 Lines

<!DOCTYPE html>
<html>
<head>
  <title>My Dashboard</title>
  <script src="h6x-runtime.js"></script>
</head>
<body>

<h6x-app name="Dashboard" mode="deterministic">
  
  <!-- Define your data -->
  <h6x-data name="users">
    [
      {"id": 1, "name": "Alice", "active": true},
      {"id": 2, "name": "Bob", "active": false}
    ]
  </h6x-data>

  <!-- Layout it out -->
  <h6x-layout type="dashboard">
    <h6x-panel slot="main">
      
      <!-- Show statistics -->
      <h6x-atom type="stat" compute="users.count" label="Total Users">
      </h6x-atom>
      
      <!-- Display the data -->
      <h6x-atom type="table" source="users">
      </h6x-atom>
      
    </h6x-panel>
  </h6x-layout>

</h6x-app>

</body>
</html>

That's it! Open in browser and it just works. No npm install, no webpack, no build process.
